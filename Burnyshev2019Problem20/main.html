def decom(data, time):
    X = np.matrix(scipy.linalg.hankel(data, np.zeros(time))).T[:, :len(data)-time+1]# 
    d = np.linalg.matrix_rank(X)#  
    S = X * X.T
    U, s, V = np.linalg.svd(S, full_matrices=True) # svd  S
    U = np.matrix(U)
    s = np.sqrt(s)
    V = np.matrix(V)
    V_ = []
    X_ = []
    for i in range(d):
        V_.append(X.T * U[:,i]/np.sqrt(s[i]))
        X_.append(np.sqrt(s[i])*U[:,i]*V_[i].T)
    return X, U, s, X_ #   , 
                    #  
                    #   X

def henkelization(x):
    x1, x2 = min(x.shape[0],x.shape[1]), max(x.shape[0],x.shape[1])
    result = []
    for i in range(1-x2, x1):
        m = np.eye(x2,k=i,dtype='bool')[::-1][:x1,:]
        mas = np.ma.masked_array(x,mask = 1-m)
        result = result + [mas.sum()/sum(sum(m))]
    return result

def prediction(series,X, number,duracity, U, s):
    days = duracity
    bas = {} #     
    for i in range(s.shape[0]):
        bas[i] = U[:,i]
    basis = {}
    for i in range(number):#   
        basis[i] = bas[i]
    coef = 0
    R = np.zeros(basis[0].shape)[:-1]#  
    for i in basis.values(): #   
        p = i[-1]
        coef = coef + p**2
        R += i[:-1]* p
    R = np.matrix(R/(1-coef))
    result = np.ravel(series.values)
    for i in range(series.shape[0], series.shape[0]+days):#   
        res = R.T * np.matrix(result[max(0, i - R.shape[0]) : i]).T
        result = np.append(result, res[0])
    return result
